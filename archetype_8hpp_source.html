194  auto add_block(std::size_t offset, const component_meta& meta) -> std::size_t {
195  const std::size_t size_in_bytes = _max_size * meta.type->size;
196  const std::size_t align = meta.type->align;
197 
198  _blocks.emplace(meta.id, offset, meta);

200  offset += detail::mod_2n(offset, align) + size_in_bytes;

205  // Calculates the maximum size of individual components this chunk buffer can hold
206  static auto get_max_size(auto&& components_meta) -> std::size_t {
207  // Calculate size of the following structure:
208  //
209  // struct {
210  // component_a_t a;
211  // component_b_t b;
212  // component_c_t c;
213  // ...
214  // };
215  //
216  auto aligned_size = aligned_components_size(components_meta);
217 
218  // handle subtraction overflow - chunk size is insufficient to hold at least one such entity
219  if (aligned_size > chunk::chunk_bytes) [[unlikely]] {
220  throw insufficient_chunk_size{ aligned_size, chunk::chunk_bytes };
221  }
222 
223  // Remaining size for packed components
224  auto remaining_space = chunk::chunk_bytes - aligned_size;
225 
226  // Calculate how much components we can pack into remaining space
227  auto remaining_elements_count = remaining_space / packed_components_size(components_meta);
228 
229  // The maximum amount of entities we can hold is grater by 1 for which we calculated aligned_size
230  return remaining_elements_count + 1;
231  }
232 
233  // Calculate size of packed structure of components
234  static auto packed_components_size(auto&& components_meta) noexcept -> std::size_t {
235  return std::accumulate(components_meta.begin(),
236  components_meta.end(),
237  component_meta::of<entity>().type->size,
238  [](const auto& res, const auto& meta) { return res + meta.type->size; });
239  }
240 
241  // Calculate size of properly aligned structure of components
242  static auto aligned_components_size(auto&& components_meta) noexcept -> std::size_t {
243  auto begin = chunk::alloc_alignment;
244  auto end = begin;
245 
246  // Add single component element size accounting for its alignment
247  auto add_elements = [&end](const component_meta& meta) {
248  end += detail::mod_2n(std::bit_cast<std::size_t>(end), meta.type->align);
249  end += meta.type->size;
250  };
251 
252  add_elements(component_meta::of<entity>());
253  for (const auto& meta : components_meta) {
254  add_elements(meta);
255  }
256 
257  return end - begin;
258  }
